    <script>

        document.getElementById('mentorshipForm').addEventListener('submit', function(event) {

            event.preventDefault();


            // Basic validation

            const expertiseCheckboxes = document.querySelectorAll('input[name="expertise"]:checked');

            if (expertiseCheckboxes.length === 0) {

                alert('Please select at least one area of mentorship expertise.');

                return;

            }


            const willing = document.querySelector('input[name="willing"]:checked');

            if (willing && willing.value === 'no') {

                alert('You must be willing to mentor to submit this application.');

                return;

            }


            // Collect form data

            const formData = new FormData(event.target);

            const data = {};

            formData.forEach((value, key) => {

                if (key === 'expertise') {

                    if (!data[key]) data[key] = [];

                    data[key].push(value);

                } else {

                    data[key] = value;

                }

            });


            // For demonstration, log the data to console

            console.log('Form Submitted:', data);

            alert('Thank you for your application! We will contact you soon.');


            // Reset form

            event.target.reset();

        });

    </script>
